package main

import (
	"fmt"
)

func longestValidParentheses(s string) int {
	if len(s) == 0 {
		return 0
	}

	// dp[i] represents the length of the longest valid parentheses ending at index i
	dp := make([]int, len(s))
	maxLen := 0

	for i := 1; i < len(s); i++ {
		if s[i] == ')' {
			if s[i-1] == '(' {
				// Case: ...()
				if i >= 2 {
					dp[i] = dp[i-2] + 2
				} else {
					dp[i] = 2
				}
			} else if dp[i-1] > 0 {
				// Case: ...))
				// We need to check if there's a matching '(' for the current ')'
				matchIdx := i - dp[i-1] - 1
				if matchIdx >= 0 && s[matchIdx] == '(' {
					dp[i] = dp[i-1] + 2
					// Add the length of valid parentheses before the matched pair
					if matchIdx > 0 {
						dp[i] += dp[matchIdx-1]
					}
				}
			}
			maxLen = max(maxLen, dp[i])
		}
	}

	return maxLen
}

func max(a, b int) int {
	if a > b {
		return a
	}
	return b
}

func main() {
	testCases := []struct {
		input    string
		expected int
	}{
		{"(()", 2},
		{")()())", 4},
		{"", 0},
		{"()", 2},
		{"(())", 4},
		{"()(())", 6},
		{")()())", 4},
		{"(()())", 6},
		{"(()()", 4},
		{"(())())", 6},
	}

	fmt.Println("Testing longestValidParentheses function:")
	for _, tc := range testCases {
		result := longestValidParentheses(tc.input)
		status := "✓"
		if result != tc.expected {
			status = "✗"
		}
		fmt.Printf("%s Input: %q, Expected: %d, Got: %d\n", status, tc.input, tc.expected, result)
	}
}
